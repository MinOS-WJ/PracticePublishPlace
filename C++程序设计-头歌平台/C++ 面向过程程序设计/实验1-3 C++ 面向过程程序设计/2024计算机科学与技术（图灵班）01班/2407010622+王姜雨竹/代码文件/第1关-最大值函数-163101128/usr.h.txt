int Max(int a,int b)
{
    /********* Begin *********/
    //找出两个参数中最大的那个并返回
    return a>b?a:b;

    
    
    /********* End *********/
}

int Max(int a,int b,int c)
{
    /********* Begin *********/
    //找出三个参数中最大的那个并返回
    return (a>b?a:b)>c?(a>b?a:b):c;

    
    
    /********* End *********/
}
